from qdrant_client.models import Filter, FieldCondition, MatchValue

async def dense_search(query_vector: list[float],
                       project_id: str,
                       top_k: int = 150) -> list[ScoredPoint]:
    return client.search(
        collection_name="document_chunks",
        query_vector=query_vector,
        query_filter=Filter(must=[
            FieldCondition(key="project_id",
                           match=MatchValue(value=project_id))
        ]),
        limit=top_k,
        with_payload=True,
        with_vectors=False
    )

2.1
BM25Searcher class
Load BM25 index from Redis. Run tokenised keyword search. Return ranked (chunk_id, bm25_score) list. Scope is always per project_id.
import pickle, redis
from rank_bm25 import BM25Okapi

class BM25Searcher:
    def __init__(self): self.redis = redis.Redis(host="redis", port=6379)

    async def search(self, query: str, project_id: str, top_k: int = 150):
        raw = self.redis.get(f"bm25:{project_id}")
        if not raw: return []
        data = pickle.loads(raw)
        bm25 = BM25Okapi([t.split() for t in data["corpus"]])
        scores = bm25.get_scores(query.split())
        ranked = sorted(zip(data["chunk_ids"], data["corpus"], scores),
                        key=lambda x: x[2], reverse=True)[:top_k]
        return [{"chunk_id": cid, "text": txt, "bm25_score": sc}
                for cid, txt, sc in ranked]
2.2
HybridRetriever — asyncio.gather for true parallelism
BM25 and dense run concurrently. Both are scoped to project_id. Returns two separate scored lists for RRF.
class HybridRetriever:
    def __init__(self):
        self.bm25   = BM25Searcher()
        self.vector = VectorSearcher()

    async def retrieve(self, query: str, query_vector: list[float],
                       project_id: str, top_k: int = 150):
        bm25_task  = asyncio.create_task(
            self.bm25.search(query, project_id, top_k))
        dense_task = asyncio.create_task(
            self.vector.dense_search(query_vector, project_id, top_k))
        bm25_hits, dense_hits = await asyncio.gather(bm25_task, dense_task)
        return bm25_hits, dense_hits
2.3
RRFMerger — Reciprocal Rank Fusion with k=60
Score(d) = Σ 1/(k + rank). Deduplicates chunk_ids across BM25 and dense lists. Returns unified ranked list with rrf_score. Fetch full payloads from Qdrant for all merged IDs in one batch call.
from collections import defaultdict

class RRFMerger:
    @staticmethod
    def merge(bm25_hits: list, dense_hits: list,
              k: int = 60) -> list[dict]:
        scores = defaultdict(float)
        for rank, h in enumerate(bm25_hits):
            scores[h["chunk_id"]] += 1 / (k + rank + 1)
        for rank, h in enumerate(dense_hits):
            scores[h.id] += 1 / (k + rank + 1)
        ranked = sorted(scores.items(), key=lambda x: x[1], reverse=True)

        # batch-fetch payloads from Qdrant for all merged IDs
        ids = [cid for cid, _ in ranked]
        payloads = client.retrieve("document_chunks", ids, with_payload=True)
        payload_map = {str(p.id): p.payload for p in payloads}

        # build max similarity score for Gate 1 from dense_hits
        dense_score_map = {str(h.id): h.score for h in dense_hits}

        return [
            RetrievedChunk(
                chunk_id        = cid,
                rrf_score       = rrf_sc,
                similarity_score= dense_score_map.get(cid, 0.0),
                **payload_map.get(cid, {})
            )
            for cid, rrf_sc in ranked
        ]

async def query_pipeline(query, query_vector, project_id) -> RetrievalResult | RefusalEvent:
    t0 = time.perf_counter()

    # Step 1: parallel retrieval
    bm25_hits, dense_hits = await HybridRetriever.retrieve(
        query, query_vector, project_id, top_k=150)

    # Step 2: RRF merge
    merged = RRFMerger.merge(bm25_hits, dense_hits, k=60)

    # Step 3: Gate 1 — score threshold (YOUR code)
    max_sim = dense_hits[0].score if dense_hits else 0.0
    if max_sim < GATE1_THRESHOLD:   # env GATE1_THRESHOLD default=0.5
        return RefusalEvent(level=1, reason="level_1_threshold",
                            message="Not found in the document.",
                            debug={"max_similarity": max_sim})

    # Step 4: neural reranking (top 150 → top 20)
    reranked = NeuralReranker.rerank(query, merged[:150])

    # Step 5: Gate 2 — reranker score (YOUR code)
    top_rr_score = reranked[0].reranker_score if reranked else 0.0
    if top_rr_score < GATE2_THRESHOLD:  # env GATE2_THRESHOLD, calibrate!
        return RefusalEvent(level=2, reason="level_2_reranker",
                            message="Not found in the document.",
                            debug={"reranker_score": top_rr_score})

    # Step 6: return — M4 takes over (Gate 3 LLM grader, PAL, Generation, Gate 4 GLEAN)
    return RetrievalResult(
        chunks       = reranked[:20],
        gate1_score  = max_sim,
        gate2_score  = top_rr_score,
        retrieval_ms = int((time.perf_counter() - t0) * 1000),
        debug        = build_debug(bm25_hits, dense_hits, merged, reranked)
    )

Gate 1 — similarity score threshold

GATE1_THRESHOLD = float(os.getenv("GATE1_THRESHOLD", "0.5"))

class RefusalGate:
    @staticmethod
    def check_threshold(dense_hits) -> RefusalEvent | None:
        max_score = dense_hits[0].score if dense_hits else 0.0
        if max_score < GATE1_THRESHOLD:
            return RefusalEvent(
                level=1, reason="level_1_threshold",
                message="Not found in the document.",
                debug={"max_similarity": max_score,
                       "threshold": GATE1_THRESHOLD}
            )
        return None  # pass → continue to reranker

4.1
NeuralReranker class — CrossEncoder batch inference
Use MiniLM-L-6-v2 (fast, CPU-friendly) as default. Zerank-1 if accuracy target not met. Model is loaded once at startup and cached in memory. Do NOT reload per request.
from sentence_transformers import CrossEncoder

class NeuralReranker:
    def __init__(self, model_name="cross-encoder/ms-marco-MiniLM-L-6-v2"):
        self.model = CrossEncoder(model_name)  # load once at startup

    def rerank(self, query: str,
               chunks: list[RetrievedChunk],
               top_k: int = 20) -> list[RetrievedChunk]:
        if not chunks: return []
        pairs  = [(query, c.raw_text) for c in chunks]
        scores = self.model.predict(pairs)   # numpy array, ~200ms for 150 pairs
        ranked = sorted(zip(chunks, scores.tolist()),
                        key=lambda x: x[1], reverse=True)
        for chunk, score in ranked[:top_k]:
            chunk.reranker_score = float(score)
        return [c for c, _ in ranked[:top_k]]

"""VectorSearcher — dense Qdrant search, always scoped to project_id."""
import os
from qdrant_client import QdrantClient
from qdrant_client.models import Filter, FieldCondition, MatchValue, SearchRequest
from backend.shared.types import RetrievedChunk, BoundingBox

client = QdrantClient(host=os.getenv("QDRANT_HOST", "qdrant"), port=6333)
COLLECTION = "document_chunks"


class VectorSearcher:
    async def dense_search(
        self, query_vector: list[float], project_id: str, top_k: int = 150
    ) -> list:
        """
        Returns List[ScoredPoint] — raw Qdrant results.
        project_id filter is MANDATORY. Never run without it.
        """
        results = client.search(
            collection_name=COLLECTION,
            query_vector=query_vector,
            query_filter=Filter(must=[
                FieldCondition(key="project_id", match=MatchValue(value=project_id))
            ]),
            limit=top_k,
            with_payload=True,
            with_vectors=False,
        )
        return results  # List[ScoredPoint] — RRFMerger reads .id, .score, .payload

    async def batch_fetch_payloads(self, chunk_ids: list[str]) -> dict[str, dict]:
        """Batch-fetch payloads for a list of chunk_ids. Used by RRFMerger."""
        points = client.retrieve(
            collection_name=COLLECTION,
            ids=chunk_ids,
            with_payload=True,
            with_vectors=False,
        )
        return {str(p.id): p.payload for p in points}

"""BM25Searcher — keyword search over Redis-backed per-project index."""
import os, pickle
import redis as redis_lib
from rank_bm25 import BM25Okapi

_redis = redis_lib.Redis(
    host=os.getenv("REDIS_HOST", "redis"),
    port=int(os.getenv("REDIS_PORT", 6379)),
    decode_responses=False
)


class BM25Searcher:
    async def search(
        self, query: str, project_id: str, top_k: int = 150
    ) -> list[dict]:
        """
        Returns List[{chunk_id, bm25_score, text}] sorted descending.
        BM25 index written by M2's DualIndexer at ingestion time.
        Redis key contract with M2: "bm25:{project_id}"
        Value: pickle({"corpus": List[str], "chunk_ids": List[str]})
        """
        raw = _redis.get(f"bm25:{project_id}")
        if not raw:
            return []  # no index yet — M2 ingestion not done

        data = pickle.loads(raw)
        corpus: list[str] = data["corpus"]
        chunk_ids: list[str] = data["chunk_ids"]

        tokenised = [doc.lower().split() for doc in corpus]
        bm25 = BM25Okapi(tokenised)
        scores = bm25.get_scores(query.lower().split())

        ranked = sorted(
            zip(chunk_ids, corpus, scores),
            key=lambda x: x[2], reverse=True
        )[:top_k]

        return [
            {"chunk_id": cid, "text": txt, "bm25_score": float(sc)}
            for cid, txt, sc in ranked
        ]

"""HybridRetriever — runs BM25 + dense in true parallel via asyncio.gather."""
import asyncio
from .bm25_searcher import BM25Searcher
from .vector_searcher import VectorSearcher


class HybridRetriever:
    def __init__(self):
        self.bm25   = BM25Searcher()
        self.vector = VectorSearcher()

    async def retrieve(
        self,
        query: str,
        query_vector: list[float],
        project_id: str,
        top_k: int = 150,
    ) -> tuple[list, list]:
        """
        Runs BM25 and dense search CONCURRENTLY.
        Total time ≈ max(bm25_time, dense_time) NOT their sum.
        Returns: (bm25_hits, dense_hits)
        """
        bm25_task  = asyncio.create_task(
            self.bm25.search(query, project_id, top_k)
        )
        dense_task = asyncio.create_task(
            self.vector.dense_search(query_vector, project_id, top_k)
        )
        bm25_hits, dense_hits = await asyncio.gather(bm25_task, dense_task)
        return bm25_hits, dense_hits

"""RRFMerger — Reciprocal Rank Fusion with k=60, dedup, batch payload hydration."""
from collections import defaultdict
from .vector_searcher import VectorSearcher
from backend.shared.types import RetrievedChunk, BoundingBox

_vs = VectorSearcher()


class RRFMerger:
    @staticmethod
    async def merge(
        bm25_hits: list[dict],
        dense_hits: list,           # List[ScoredPoint]
        k: int = 60,
    ) -> list[RetrievedChunk]:
        """
        Formula:  Score(d) = Σ  1 / (k + rank(d))
        k=60 is the smoothing constant from the spec — do NOT change.
        Steps:
          1. Score both ranked lists
          2. Deduplicate by chunk_id
          3. Sort by merged score descending
          4. Batch-fetch payloads from Qdrant (single round-trip)
          5. Build RetrievedChunk objects
        """
        scores: dict[str, float] = defaultdict(float)

        for rank, hit in enumerate(bm25_hits):
            scores[hit["chunk_id"]] += 1.0 / (k + rank + 1)

        dense_score_map: dict[str, float] = {}
        for rank, hit in enumerate(dense_hits):
            cid = str(hit.id)
            scores[cid] += 1.0 / (k + rank + 1)
            dense_score_map[cid] = hit.score   # raw cosine — Gate 1 needs this

        ranked = sorted(scores.items(), key=lambda x: x[1], reverse=True)

        # Batch fetch all payloads in ONE Qdrant round-trip
        all_ids = [cid for cid, _ in ranked]
        payload_map = await _vs.batch_fetch_payloads(all_ids)

        bm25_score_map = {h["chunk_id"]: h["bm25_score"] for h in bm25_hits}

        result: list[RetrievedChunk] = []
        for cid, rrf_sc in ranked:
            pl = payload_map.get(cid, {})
            bb_raw = pl.get("bounding_box")
            bb = BoundingBox(**bb_raw) if bb_raw else None
            result.append(RetrievedChunk(
                chunk_id        = cid,
                project_id      = pl.get("project_id", ""),
                file_id         = pl.get("file_id", ""),
                page_number     = pl.get("page_number", 0),
                section_header  = pl.get("section_header", ""),
                chunk_index     = pl.get("chunk_index", 0),
                raw_text        = pl.get("raw_text", ""),
                context_summary = pl.get("context_summary", ""),
                bounding_box    = bb,
                is_table        = pl.get("is_table", False),
                table_html      = pl.get("table_html"),
                token_count     = pl.get("token_count", 0),
                similarity_score= dense_score_map.get(cid, 0.0),
                bm25_score      = bm25_score_map.get(cid, 0.0),
                rrf_score       = rrf_sc,
            ))
        return result

"""NeuralReranker — CrossEncoder batch inference. Model loaded ONCE at startup."""
import os
from sentence_transformers import CrossEncoder
from backend.shared.types import RetrievedChunk

_MODEL_NAME = os.getenv(
    "RERANKER_MODEL", "cross-encoder/ms-marco-MiniLM-L-6-v2"
)
# Loaded once at module import — NOT per request
_model: CrossEncoder | None = None


def _get_model() -> CrossEncoder:
    global _model
    if _model is None:
        _model = CrossEncoder(_MODEL_NAME)   # ~60MB, downloads to Docker volume
    return _model


class NeuralReranker:
    def rerank(
        self,
        query: str,
        chunks: list[RetrievedChunk],
        top_k: int = 20,
    ) -> list[RetrievedChunk]:
        """
        Runs CrossEncoder on (query, chunk.raw_text) pairs.
        Scores ~150 pairs in ~250ms on CPU (MiniLM-L-6-v2).
        Sets chunk.reranker_score in-place.
        Returns top_k chunks sorted by reranker_score descending.
        """
        if not chunks:
            return []

        model = _get_model()
        pairs  = [(query, c.raw_text) for c in chunks]
        scores = model.predict(pairs).tolist()   # numpy → python list

        for chunk, score in zip(chunks, scores):
            chunk.reranker_score = float(score)

        ranked = sorted(chunks, key=lambda c: c.reranker_score, reverse=True)
        return ranked[:top_k]

"""RefusalGate — Gate 1 (similarity threshold) and Gate 2 (reranker threshold).
Gates 3 and 4 are owned by M4 (reasoning_engine.py and glean_verifier.py).
"""
import os
from backend.shared.types import RetrievedChunk, RefusalEvent

GATE1_THRESHOLD = float(os.getenv("GATE1_THRESHOLD", "0.5"))
GATE2_THRESHOLD = float(os.getenv("GATE2_THRESHOLD", "0.0"))  # calibrate in Phase 4!


class RefusalGate:
    @staticmethod
    def check_threshold(dense_hits: list) -> RefusalEvent | None:
        """
        Gate 1: fires if max dense cosine similarity < GATE1_THRESHOLD.
        Short-circuits the entire pipeline — no reranking, no LLM call.
        Cost: ~0ms. Use it to block hallucinations cheaply.
        """
        max_score = dense_hits[0].score if dense_hits else 0.0
        if max_score < GATE1_THRESHOLD:
            return RefusalEvent(
                level  = 1,
                reason = "level_1_threshold",
                message= "Not found in the document.",
                debug  = {
                    "max_similarity": round(max_score, 4),
                    "threshold":      GATE1_THRESHOLD,
                }
            )
        return None  # pass — continue to reranker

    @staticmethod
    def check_reranker(reranked: list[RetrievedChunk]) -> RefusalEvent | None:
        """
        Gate 2: fires if top reranker score < GATE2_THRESHOLD.
        Threshold MUST be calibrated (Phase 4) — default 0.0 means gate never fires.
        Run adversarial query set to find correct threshold.
        """
        top = reranked[0].reranker_score if reranked else 0.0
        if top < GATE2_THRESHOLD:
            return RefusalEvent(
                level  = 2,
                reason = "level_2_reranker",
                message= "Not found in the document.",
                debug  = {
                    "reranker_score": round(top, 4),
                    "threshold":      GATE2_THRESHOLD,
                }
            )
        return None  # pass — return RetrievalResult to M3

async def query_pipeline(
    query:      str,
    query_vector: list[float],
    project_id: str,
    top_k:      int = 150,
) -> RetrievalResult | RefusalEvent:
    """
    Full retrieval pipeline. Returns RetrievalResult on success or RefusalEvent on gate fire.
    M3 checks isinstance(result, RefusalEvent) to decide whether to continue or emit SSE refusal.
    """
    t0 = time.perf_counter()

    # ── Step 1: parallel BM25 + dense ─────────────────────────────────────
    bm25_hits, dense_hits = await _hybrid.retrieve(
        query, query_vector, project_id, top_k=top_k
    )

    # ── Step 2: Gate 1 — score threshold (no LLM, no reranker) ───────────
    gate1 = RefusalGate.check_threshold(dense_hits)
    if gate1:
        gate1.debug["retrieval_ms"] = int((time.perf_counter() - t0) * 1000)
        return gate1

    # ── Step 3: RRF merge ─────────────────────────────────────────────────
    merged = await RRFMerger.merge(bm25_hits, dense_hits, k=60)

    # ── Step 4: Neural reranking (top_k → top 20) ─────────────────────────
    reranked = _reranker.rerank(query, merged[:top_k], top_k=20)

    # ── Step 5: Gate 2 — reranker threshold ───────────────────────────────
    gate2 = RefusalGate.check_reranker(reranked)
    if gate2:
        gate2.debug["retrieval_ms"] = int((time.perf_counter() - t0) * 1000)
        return gate2

    # ── Step 6: Return result → M3 hands off to M4 ────────────────────────
    retrieval_ms = int((time.perf_counter() - t0) * 1000)
    return RetrievalResult(
        chunks      = reranked,
        gate1_score = dense_hits[0].score if dense_hits else 0.0,
        gate2_score = reranked[0].reranker_score if reranked else 0.0,
        retrieval_ms= retrieval_ms,
        debug       = build_debug(bm25_hits, dense_hits, merged, reranked),
    )

import json, asyncio, argparse, os, sys
from pathlib import Path

ROOT = Path(__file__).parent.parent.parent
sys.path.insert(0, str(ROOT))

from backend.retrieval import query_pipeline
from openai import AsyncOpenAI

_oa = AsyncOpenAI(api_key=os.getenv("OPENAI_API_KEY"))


async def embed(text: str) -> list[float]:
    r = await _oa.embeddings.create(input=text, model="text-embedding-3-large")
    return r.data[0].embedding


async def run_eval(dataset_path: str, mode: str = "post-rerank") -> dict:
    with open(dataset_path) as f:
        dataset = json.load(f)

    # Only evaluate non-adversarial queries for P@k / MRR / NDCG
    eval_rows  = [r for r in dataset if not r.get("adversarial")]
    adv_rows   = [r for r in dataset if r.get("adversarial")]

    # ── Build ranx Qrels ──────────────────────────────────────────────────
    qrels_dict = {}
    for row in eval_rows:
        qrels_dict[row["id"]] = {cid: 1 for cid in row["relevant_ids"]}

    # ── Run retrieval ─────────────────────────────────────────────────────
    run_dict = {}
    PROJECT_ID = os.getenv("EVAL_PROJECT_ID", "adani-q2-fy26")

    for row in eval_rows:
        vector = await embed(row["query"])
        result = await query_pipeline(row["query"], vector, PROJECT_ID)
        if hasattr(result, "chunks"):
            run_dict[row["id"]] = {
                c.chunk_id: c.reranker_score if mode == "post-rerank" else c.rrf_score
                for c in result.chunks
            }
        else:
            # gate fired — no chunks retrieved
            run_dict[row["id"]] = {}

    # ── Adversarial gate test ─────────────────────────────────────────────
    adv_pass = 0
    for row in adv_rows:
        vector = await embed(row["query"])
        result = await query_pipeline(row["query"], vector, PROJECT_ID)
        if hasattr(result, "level"):   # RefusalEvent
            adv_pass += 1

    # ── Compute metrics ───────────────────────────────────────────────────
    from ranx import Run, Qrels, evaluate

    qrels = Qrels(qrels_dict)
    run   = Run(run_dict)
    metrics = evaluate(qrels, run, ["precision@5", "precision@10", "mrr", "ndcg@10"])

    output = {
        "mode":           mode,
        "eval_rows":      len(eval_rows),
        "adv_rows":       len(adv_rows),
        "adv_pass_rate":  round(adv_pass / len(adv_rows), 3) if adv_rows else 1.0,
        "precision@5":    round(metrics["precision@5"], 4),
        "precision@10":   round(metrics["precision@10"], 4),
        "mrr":            round(metrics["mrr"], 4),
        "ndcg@10":        round(metrics["ndcg@10"], 4),
    }
    print(json.dumps(output, indent=2))
    return output


if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--mode", default="post-rerank",
                        choices=["post-rerank", "pre-rerank"])
    parser.add_argument("--assert-ndcg", type=float, default=None)

GOOD_QUERIES = [
    "What are the major business segments?",
    "What is consolidated total income H1-26?",
    "What drove EBITDA growth in H1-26?",
    "Describe airport performance in H1-26",
    "What is total revenue from energy segment?",

    "What is the CEO's email address?",
    "What is the chairman's phone number?",

    "What will Adani do in 2035?",
    "Summarise slide 47",
    "What is the weather in Mumbai today?",

    for q in GOOD_QUERIES:
        v = await embed(q)
        r = await query_pipeline(q, v, PROJECT_ID)
        if hasattr(r, "chunks") and r.chunks:
            good_scores.append(r.chunks[0].reranker_score)

    for q in BAD_QUERIES:

        score = r.chunks[0].reranker_score if hasattr(r, "chunks") and r.chunks else 0.0
        bad_scores.append(score)

    all_scores = sorted(set(good_scores + bad_scores))
    for t in all_scores:
        fpr = sum(1 for s in good_scores if s < t) / len(good_scores)
        if fpr < 0.05:
